	drop table agent_object;
	drop table fund_object;

	drop table action_object;

	drop view fund_object_v;
	drop view fund_alloc_v;
	drop view action_object_v;
	drop view action_partis_v;

	create table agent_object(
		id char(32) primary key,
		name char(30) not null,
		display varchar(200),
		`key` varchar(40) not null,
		`call` char(200)
	);

	create table fund_object(
		id char(32) primary key,
		name varchar(200) not null,
		`usage` varchar(200) not null,
		admin char(32) not null
	);

	create table fund_alloc(
		id char(32) primary key,
		fund char(32) not null,
		agent char(32) not null,
		role char(30),
		quantity double not null
	);

	create table action_object(
		id char(32) primary key,
		name varchar(200) not null,
		class varchar(200),
		fund char(32) not null,
		cost double not null
	);

	create table action_partis(
		id char(32) primary key,
		action char(32) not null,
		agent char(32) not null,
		adjust char(25)
	);

	create table touch_diary(
		id char(32) primary key,
		obj char(32) not null,
		objt char(32) not null,
		time char(30) not null
	);

	create table system_diary(
		id char(32) primary key,
		time char(30) not null,
		content text
	);


	create view fund_object_v as
		select f.id, f.name as fund_name, f.`usage`, a.name as agent_name
		from fund_object f left join agent_object a on f.admin=a.id
		;

	create view fund_alloc_v as
		select l.id, f.name as fund_name, a.name as agent_name, role, quantity
		from fund_alloc l left join fund_object f on l.fund=f.id
			left join agent_object a on l.agent=a.id
		;

	create view action_object_v as
		select c.id, c.name, c.class, f.name as fund_name, c.cost
		from action_object c left join fund_object f on c.fund=f.id
		;

	create view action_partis_v as
		select p.id, c.name as action_name, a.name as agent_name, adjust
		from action_partis p left join action_object c on p.action=c.id
			left join agent_object a on p.agent=a.id
		;
